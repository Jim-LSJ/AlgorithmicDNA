For binary Tree, DFS only has three templates
Inorder traversal: left -> current -> right
Preorder traversal: current -> left -> right
Postorder traversal: left -> right -> current
DFS in binary tree does not require a visited set or list because binary tree is a directed acyclic graph (DAG). 
A visited set or list is required if cycles might exist.

class SegmentTree:	def __init__(self, data):		self.n = len(data)		self.tree = [0] * (4*self.n)		self._build(data, 1, 0, self.n-1)	def _build(self, data, node, left, right):		if left == right:			self.tree[node] = data[left]		else:			mid = left + (right - left) // 2			self._build(data, node*2, left, mid)			self._build(data, node*2+1, mid+1, right)			self.tree[node] = self.tree[node*2] + self.tree[node*2+1]		def query(self, left, right):		return self._query(1, 0, self.n-1, left, right)		def _query(self, node, l, r, left, right):		if right < l or left > r:			return 0		if left <= l and r <= right:			return self.tree[node]				mid = (l + r) // 2		return (self._query(node*2, l, mid, left, right) + 		              self._query(node*2 + 1, mid+1, r, left, right))
